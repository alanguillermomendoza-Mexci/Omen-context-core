import sqlite3
import os
import sys
from datetime import datetime

# ============================================================
# CONFIGURACIÓN
# ============================================================

DB_FILENAME = "omen_context.db"
TIMESTAMP   = datetime.now().strftime("%Y-%m-%d %H:%M:%S")


# ============================================================
# ESQUEMA SQL
# ============================================================

SCHEMA_SQL = """
PRAGMA foreign_keys = ON;

CREATE TABLE IF NOT EXISTS projects (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL UNIQUE,
    description TEXT,
    status_id   INTEGER,
    notes       TEXT,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now', 'localtime')),
    updated_at  TEXT,
    FOREIGN KEY (status_id) REFERENCES statuses(id)
);

CREATE TABLE IF NOT EXISTS entry_types (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL UNIQUE,
    description TEXT,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now', 'localtime'))
);

CREATE TABLE IF NOT EXISTS priorities (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL UNIQUE,
    description TEXT,
    sort_order  INTEGER NOT NULL DEFAULT 0,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now', 'localtime'))
);

CREATE TABLE IF NOT EXISTS statuses (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL UNIQUE,
    description TEXT,
    sort_order  INTEGER NOT NULL DEFAULT 0,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now', 'localtime'))
);

CREATE TABLE IF NOT EXISTS modules (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id  INTEGER NOT NULL,
    name        TEXT    NOT NULL,
    description TEXT,
    status_id   INTEGER,
    priority_id INTEGER,
    notes       TEXT,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now', 'localtime')),
    updated_at  TEXT,
    FOREIGN KEY (project_id)  REFERENCES projects(id),
    FOREIGN KEY (status_id)   REFERENCES statuses(id),
    FOREIGN KEY (priority_id) REFERENCES priorities(id),
    UNIQUE (project_id, name)
);

CREATE TABLE IF NOT EXISTS context_entries (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id      INTEGER NOT NULL,
    module_id       INTEGER NOT NULL,
    entry_type_id   INTEGER NOT NULL,
    priority_id     INTEGER,
    status_id       INTEGER,
    title           TEXT    NOT NULL CHECK (trim(title) != ''),
    summary         TEXT,
    content         TEXT    NOT NULL CHECK (trim(content) != ''),
    source          TEXT,
    event_datetime  TEXT,
    is_key_context  INTEGER NOT NULL DEFAULT 0 CHECK (is_key_context IN (0, 1)),
    notes           TEXT,
    created_at      TEXT    NOT NULL DEFAULT (datetime('now', 'localtime')),
    updated_at      TEXT,
    FOREIGN KEY (project_id)    REFERENCES projects(id),
    FOREIGN KEY (module_id)     REFERENCES modules(id),
    FOREIGN KEY (entry_type_id) REFERENCES entry_types(id),
    FOREIGN KEY (priority_id)   REFERENCES priorities(id),
    FOREIGN KEY (status_id)     REFERENCES statuses(id)
);

CREATE TABLE IF NOT EXISTS tags (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL UNIQUE,
    description TEXT,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now', 'localtime'))
);

CREATE TABLE IF NOT EXISTS entry_tags (
    id         INTEGER PRIMARY KEY AUTOINCREMENT,
    entry_id   INTEGER NOT NULL,
    tag_id     INTEGER NOT NULL,
    created_at TEXT    NOT NULL DEFAULT (datetime('now', 'localtime')),
    FOREIGN KEY (entry_id) REFERENCES context_entries(id),
    FOREIGN KEY (tag_id)   REFERENCES tags(id),
    UNIQUE (entry_id, tag_id)
);

CREATE TABLE IF NOT EXISTS entry_relations (
    id               INTEGER PRIMARY KEY AUTOINCREMENT,
    source_entry_id  INTEGER NOT NULL,
    target_entry_id  INTEGER NOT NULL,
    relation_type    TEXT    NOT NULL,
    description      TEXT,
    created_at       TEXT    NOT NULL DEFAULT (datetime('now', 'localtime')),
    FOREIGN KEY (source_entry_id) REFERENCES context_entries(id),
    FOREIGN KEY (target_entry_id) REFERENCES context_entries(id),
    CHECK (source_entry_id != target_entry_id)
);

CREATE TABLE IF NOT EXISTS activity_log (
    id                 INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id         INTEGER,
    module_id          INTEGER,
    entry_id           INTEGER,
    action_type        TEXT    NOT NULL,
    action_description TEXT,
    source             TEXT,
    created_at         TEXT    NOT NULL DEFAULT (datetime('now', 'localtime')),
    FOREIGN KEY (project_id) REFERENCES projects(id),
    FOREIGN KEY (module_id)  REFERENCES modules(id),
    FOREIGN KEY (entry_id)   REFERENCES context_entries(id)
);

CREATE INDEX IF NOT EXISTS idx_ce_project_id    ON context_entries(project_id);
CREATE INDEX IF NOT EXISTS idx_ce_module_id     ON context_entries(module_id);
CREATE INDEX IF NOT EXISTS idx_ce_entry_type_id ON context_entries(entry_type_id);
CREATE INDEX IF NOT EXISTS idx_ce_priority_id   ON context_entries(priority_id);
CREATE INDEX IF NOT EXISTS idx_ce_status_id     ON context_entries(status_id);
CREATE INDEX IF NOT EXISTS idx_ce_created_at    ON context_entries(created_at);
CREATE INDEX IF NOT EXISTS idx_ce_is_key        ON context_entries(is_key_context);
CREATE INDEX IF NOT EXISTS idx_mod_project_id   ON modules(project_id);
CREATE INDEX IF NOT EXISTS idx_et_entry_id      ON entry_tags(entry_id);
CREATE INDEX IF NOT EXISTS idx_et_tag_id        ON entry_tags(tag_id);
CREATE INDEX IF NOT EXISTS idx_er_source        ON entry_relations(source_entry_id);
CREATE INDEX IF NOT EXISTS idx_er_target        ON entry_relations(target_entry_id);
CREATE INDEX IF NOT EXISTS idx_al_project_id    ON activity_log(project_id);
CREATE INDEX IF NOT EXISTS idx_al_module_id     ON activity_log(module_id);
CREATE INDEX IF NOT EXISTS idx_al_entry_id      ON activity_log(entry_id);
CREATE INDEX IF NOT EXISTS idx_al_created_at    ON activity_log(created_at);
CREATE INDEX IF NOT EXISTS idx_tags_name        ON tags(name);
CREATE INDEX IF NOT EXISTS idx_etype_name       ON entry_types(name);
CREATE INDEX IF NOT EXISTS idx_pri_name         ON priorities(name);
CREATE INDEX IF NOT EXISTS idx_sta_name         ON statuses(name);
"""


# ============================================================

# ============================================================
# VERIFICACIÓN DE BASE EXISTENTE
# ============================================================

def verificar_base_existente(db_path):
    """
    Si la base de datos ya existe, pide confirmación antes de continuar.
    No destruye la base; el esquema usa IF NOT EXISTS para ser seguro.
    """
    if os.path.exists(db_path):
        print(f"\n[AVISO] El archivo '{db_path}' ya existe.")
        print("  El script NO eliminará la base de datos existente.")
        print("  Solo ejecutará CREATE IF NOT EXISTS e INSERT OR IGNORE.")
        print("  Los datos existentes se conservarán intactos.\n")
        respuesta = input("  ¿Deseas continuar con la inicialización segura? (s/n): ").strip().lower()
        if respuesta != "s":
            print("\n[CANCELADO] Operación cancelada por el usuario.")
            sys.exit(0)
        return True
    return False


# ============================================================
# CREACIÓN DE LA BASE
# ============================================================

def crear_base(db_path):
    """Crea la base de datos, aplica el esquema e inserta semilla."""

    print(f"\n[INICIO] Inicializando base de datos: {db_path}")

    con = sqlite3.connect(db_path)
    con.execute("PRAGMA foreign_keys = ON;")

    try:
        cursor = con.cursor()

        # Aplicar esquema completo
        print("  [OK] Aplicando esquema SQL...")
        con.executescript(SCHEMA_SQL)

        # Verificar si ya hay datos semilla
        resultado = cursor.execute("SELECT COUNT(*) FROM statuses").fetchone()
        if resultado[0] == 0:
            print("  [OK] Insertando datos semilla...")
            insertar_semilla(cursor, TIMESTAMP)
        else:
            print("  [INFO] Datos semilla ya presentes. No se reinsertan.")

        con.commit()
        print(f"\n[ÉXITO] Base de datos lista: {db_path}")

    except sqlite3.Error as e:
        con.rollback()
        print(f"\n[ERROR] Fallo durante la inicialización: {e}")
        sys.exit(1)

    finally:
        con.close()


# ============================================================
# REPORTE DE VERIFICACIÓN
# ============================================================

def reporte_verificacion(db_path):
    """Muestra un resumen de las tablas creadas y registros semilla."""

    print("\n--- REPORTE DE VERIFICACIÓN ---")
    con = sqlite3.connect(db_path)
    cursor = con.cursor()

    tablas = [
        "projects", "entry_types", "priorities", "statuses",
        "modules", "context_entries", "tags", "entry_tags",
        "entry_relations", "activity_log"
    ]

    for tabla in tablas:
        try:
            count = cursor.execute(f"SELECT COUNT(*) FROM {tabla}").fetchone()[0]
            print(f"  {tabla:<25} {count:>4} registro(s)")
        except sqlite3.Error as e:
            print(f"  {tabla:<25} ERROR: {e}")

    con.close()
    print("-------------------------------\n")


# ============================================================
# PUNTO DE ENTRADA
# ============================================================

def main():
    db_path = DB_FILENAME

    verificar_base_existente(db_path)
    crear_base(db_path)
    reporte_verificacion(db_path)

    print("[SISTEMA] OMEN Context Database v1.0 inicializada correctamente.")
    print("[SISTEMA] Lista para uso manual inmediato.\n")


if __name__ == "__main__":
    main()
